🔧 راهنمای UserScript - راه‌حل حرفه‌ای

✨ مزایای UserScript:
  • خودکار روی همه سایت‌های AI فعال می‌شود
  • نیازی به bookmark نیست
  • آپدیت خودکار
  • روی موبایل هم کار می‌کند

مرحله 1: نصب Tampermonkey

  1. دسکتاپ: به Chrome Web Store بروید و "Tampermonkey" را نصب کنید
  2. موبایل اندروید: Firefox + Tampermonkey addon یا Kiwi Browser + Tampermonkey
  3. موبایل iOS: Safari + Userscripts app

مرحله 2: نصب اسکریپت

  1. کد زیر را کپی کنید
  2. در Tampermonkey، "Create new script" کلیک کنید
  3. کد را paste کنید و Save کنید
  4. تمام! حالا روی همه سایت‌ها خودکار کار می‌کند
// ==UserScript==
// @name         AI RTL Persian Support
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  Auto RTL for AI platforms
// @author       Claude RTL
// @match        https://claude.ai/*
// @match        https://chatgpt.com/*
// @match        https://gemini.google.com/*
// @match        https://copilot.microsoft.com/*
// @match        https://you.com/*
// @match        https://perplexity.ai/*
// @match        https://poe.com/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';
    
    // Add Persian font CSS
    const css = \`
        @import url('https://cdn.jsdelivr.net/gh/rastikerdar/iranyekan-web-font@v1.0.0/dist/font-face.css');
        .rtl-persian {
            font-family: 'IRANYekan', Tahoma, sans-serif !important;
            direction: rtl !important;
            text-align: right !important;
        }
        .rtl-font {
            font-family: 'IRANYekan', Tahoma, sans-serif !important;
        }
        pre, code {
            direction: ltr !important;
            text-align: left !important;
        }
    \`;
    
    const style = document.createElement('style');
    style.textContent = css;
    document.head.appendChild(style);
    
    // Persian detection
    function isPersian(text) {
        if (!text) return false;
        const persianChars = (text.match(/[\u0600-\u06FF\u0750-\u077F\u08A0-\u08FF]/g) || []).length;
        const englishChars = (text.match(/[A-Za-z]/g) || []).length;
        return persianChars > englishChars * 0.7;
    }
    
    // Process elements
    function processElements() {
        document.querySelectorAll('p, div, span, h1, h2, h3, h4, h5, h6, li, textarea, [contenteditable="true"]').forEach(el => {
            if (el.closest('pre') || el.closest('code')) return;
            
            const text = el.textContent || el.value || '';
            el.classList.add('rtl-font');
            
            if (isPersian(text)) {
                el.classList.add('rtl-persian');
                el.setAttribute('dir', 'rtl');
            } else if (text.trim()) {
                el.setAttribute('dir', 'ltr');
            }
        });
    }
    
    // Initial process
    processElements();
    
    // Watch for changes
    new MutationObserver(processElements).observe(document.body, {
        childList: true,
        subtree: true
    });
    
    // Handle input events
    document.addEventListener('input', (e) => {
        const el = e.target;
        const text = el.value || el.textContent || '';
        el.classList.add('rtl-font');
        
        if (isPersian(text)) {
            el.classList.add('rtl-persian');
            el.setAttribute('dir', 'rtl');
        } else {
            el.classList.remove('rtl-persian');
            el.setAttribute('dir', 'ltr');
        }
    });
    
    console.log('✅ AI RTL UserScript loaded!');
})();

💻 اسکریپت دستی RTL

  1. به صفحه Claude بروید
  2. F12 را فشار دهید تا Developer Tools باز شود
  3. تب Console را انتخاب کنید
  4. کد زیر را کپی کرده و در Console paste کنید
  5. Enter را بزنید
// Claude RTL Script
(function() {
    'use strict';
    
    if (window.claudeRTLLoaded) {
        alert('RTL قبلاً فعال شده است!');
        return;
    }
    window.claudeRTLLoaded = true;
    
    // Add Persian font CSS
    const css = `
        @import url('https://cdn.jsdelivr.net/gh/rastikerdar/iranyekan-web-font@v1.0.0/dist/font-face.css');
        
        .rtl-persian {
            font-family: 'IRANYekan', Tahoma, sans-serif !important;
            direction: rtl !important;
            text-align: right !important;
        }
        
        .rtl-font {
            font-family: 'IRANYekan', Tahoma, sans-serif !important;
        }
        
        pre, code {
            direction: ltr !important;
            text-align: left !important;
        }
        
        .ProseMirror[dir="rtl"] {
            text-align: right !important;
        }
        
        ul[dir="rtl"], ol[dir="rtl"] {
            padding-right: 1.5rem !important;
            padding-left: 0 !important;
        }
    `;
    
    const style = document.createElement('style');
    style.textContent = css;
    document.head.appendChild(style);
    
    // Persian detection
    function isPersian(text) {
        if (!text || typeof text !== 'string') return false;
        
        const persianChars = (text.match(/[\u0600-\u06FF\u0750-\u077F\u08A0-\u08FF]/g) || []).length;
        const englishChars = (text.match(/[A-Za-z]/g) || []).length;
        
        return persianChars > englishChars * 0.7;
    }
    
    // Process elements
    function processElements() {
        // Text elements
        document.querySelectorAll('p, div, span, h1, h2, h3, h4, h5, h6, li, textarea, [contenteditable="true"]').forEach(el => {
            if (el.closest('pre') || el.closest('code')) return;
            
            const text = el.textContent || el.value || '';
            el.classList.add('rtl-font');
            
            if (isPersian(text)) {
                el.classList.add('rtl-persian');
                el.setAttribute('dir', 'rtl');
            } else if (text.trim()) {
                el.setAttribute('dir', 'ltr');
            }
        });
        
        // Special handling for ProseMirror (Claude's input)
        document.querySelectorAll('.ProseMirror').forEach(el => {
            const text = el.textContent || '';
            el.classList.add('rtl-font');
            
            if (isPersian(text)) {
                el.setAttribute('dir', 'rtl');
            } else if (text.trim()) {
                el.setAttribute('dir', 'ltr');
            } else {
                el.setAttribute('dir', 'auto');
            }
        });
    }
    
    // Initial process
    processElements();
    
    // Watch for changes
    const observer = new MutationObserver(processElements);
    observer.observe(document.body, {
        childList: true,
        subtree: true
    });
    
    // Handle input events
    document.addEventListener('input', (e) => {
        const el = e.target;
        const text = el.value || el.textContent || '';
        
        el.classList.add('rtl-font');
        
        if (isPersian(text)) {
            el.classList.add('rtl-persian');
            el.setAttribute('dir', 'rtl');
        } else {
            el.classList.remove('rtl-persian');
            el.setAttribute('dir', 'ltr');
        }
    });
    
    console.log('✅ Claude RTL successfully loaded!');
    alert('✅ Claude RTL فعال شد!\n\n• فونت فارسی اعمال شد\n• تشخیص خودکار جهت متن فعال شد\n• حالا می‌توانید از فارسی استفاده کنید');
})();
